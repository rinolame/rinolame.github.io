SELECT DISTINCT sl.suspect_id, sl.name,
    al.door_accessed, al.access_time,
    sl.alibi, cl.call_time
FROM suspects_large sl
JOIN access_logs_large al
    ON al.suspect_id = sl.suspect_id
JOIN call_records_large cl
    ON sl.suspect_id = cl.suspect_id
WHERE al.door_accessed = 'Vault Room'
  AND al.success_flag = 'TRUE'
  AND cl.recipient_relation = 'Victim'
ORDER BY al.access_time;
